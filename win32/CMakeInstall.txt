#使用windeployqt添加依赖
find_program(WINDEPLOYQT NAMES windeployqt)
if(NOT WINDEPLOYQT-NOTFOUND)
	message(STATUS "windeployqt(qt5) is found.qt executable program path is ${CMAKE_INSTALL_PREFIX}/bin")
	execute_process(COMMAND ${WINDEPLOYQT} --release --force  --compiler-runtime  --qmldir ${CMAKE_INSTALL_PREFIX}/bin  ${CMAKE_INSTALL_PREFIX}/bin)
else()
	message(STATUS "windeployqt(qt5) is not found.package maybe invalid.")
endif()
